# 无界学城穿梭巴士

深圳大学城（南山区）免费校园摆渡车，服务清华、北大、哈工大三所院校。三条线路均为环形，每班约 40 分钟跑完全程。

## 获取实时数据

将下方脚本写入临时文件后运行，可选参数为线路：`1`、`2c`（顺时针）、`2a`（逆时针），不填则查询全部。脚本无外部依赖，自动匿名登录，输出 JSON，每辆车一条：

```json
{
  "plate": "深D000047",
  "online": true,
  "current_station": "清华教学楼C栋(去程)",
  "station_sort": 8,
  "direction": "西",
  "speed_kmh": 9.0,
  "gps_time": "2026-04-09 18:04:35",
  "line": "无界学城一号线",
  "line_key": "1"
}
```

列表为空 = 当前无车运行（非运营时段或班次间隙）。

## 估算到站时间

每站约 2–3 分钟。剩余站数 = (目标 sort − 当前 sort + 总站数) % 总站数，乘以 2.5 分钟即可。

def api_post(path, data, token="", form=False):
    url = f"https://predict.ipubtrans.com{path}"
    headers = {"tenantId": TENANT_ID, "qrCodeId": QR_CODE_ID, "token": token}
    if form:
        body = urllib.parse.urlencode(data).encode()
        headers["Content-Type"] = "application/x-www-form-urlencoded;charset=UTF-8"
    else:
        body = json.dumps(data).encode()
        headers["Content-Type"] = "application/json"
    req = urllib.request.Request(url, data=body, headers=headers, method="POST")
    with urllib.request.urlopen(req, timeout=10) as resp:
        return json.loads(resp.read())

def login():
    r = api_post("/mobile/login/guise", {})
    if r["returnCode"] != 200: raise RuntimeError(r["returnInfo"])
    return r["returnData"]["token"]

def station_name_by_id(line_key, sid):
    for s in LINE_STATIONS.get(line_key, []):
        if s["id"] == sid: return f"站{s['sort']} {s['name']}"
    return f"未知站 ({sid})"

def direction_label(deg):
    if deg is None: return "?"
    return ["北","东北","东","东南","南","西南","西","西北"][round(deg/45)%8]

def get_vehicles(token, line_key):
    line = LINES[line_key]
    r = api_post("/mobile/predict/line/vehList", {"lineId": line["id"]}, token=token, form=True)
    if r["returnCode"] != 200: return []
    out = []
    for v in r.get("returnData") or []:
        g = v.get("gps") or {}
        out.append({
            "plate": v.get("vehCode","?"), "online": g.get("onlineStatus")==1,
            "lat": g.get("lat"), "lng": g.get("lng"),
            "speed_kmh": g.get("speed"), "direction": direction_label(g.get("direction")),
            "gps_time": g.get("gpsTime"),
            "current_station": station_name_by_id(line_key, v.get("stationId","")),
            "station_sort": v.get("sort"), "line": line["name"], "line_key": line_key,
        })
    return out

def main():

        print(f"Unknown line '{arg}'. Valid: {', '.join(LINES)}", file=sys.stderr); sys.exit(1)
    token = login()
    vehicles = []
    for k in ([arg] if arg else list(LINES)):
        vehicles.extend(get_vehicles(token, k))
    print(json.dumps(vehicles, ensure_ascii=False, indent=2))

if __name__ == "__main__":
    main()
